Create `SyncBuds/Shared/GlobalHotkeyManager.swift`. This class wraps Carbon's RegisterEventHotKey, which is the ONLY sandbox-compatible global hotkey API on macOS. Do NOT use NSEvent.addGlobalMonitorForEvents (requires Accessibility permission, blocked by sandbox).

CRITICAL PITFALL (Pitfall 5 from research): Carbon's InstallEventHandler requires a @convention(c) function pointer — Swift closures cannot be passed directly. The workaround is to store the action closure on a global/static and use a static C-compatible function as the bridge.

```swift
//
//  GlobalHotkeyManager.swift
//  SyncBuds
//

#if os(macOS)
import Carbon
import Foundation

/// Wraps Carbon RegisterEventHotKey for sandbox-compatible global keyboard shortcuts.
/// NSEvent.addGlobalMonitorForEvents is NOT used — it requires Accessibility permission
/// which cannot be granted to sandboxed apps.
final class GlobalHotkeyManager {

    // Static storage for the action — needed for @convention(c) C function bridge
    private static var storedAction: (() -> Void)?
    private var hotKeyRef: EventHotKeyRef?
    private var eventHandler: EventHandlerRef?

    init() {}

    deinit {
        unregister()
    }

    /// Register a global hotkey. Call whenever keyCode or modifiers change.
    /// - Parameters:
    ///   - keyCode: Virtual key code (e.g. 49 for Space)
    ///   - modifiers: Carbon modifier flags (e.g. cmdKey | shiftKey)
    ///   - action: Closure invoked when hotkey is pressed
    func register(keyCode: UInt32, modifiers: UInt32, action: @escaping () -> Void) {
        unregister()
        GlobalHotkeyManager.storedAction = action

        // Install event handler with @convention(c) bridging function
        var eventType = EventTypeSpec(
            eventClass: OSType(kEventClassKeyboard),
            eventKind: UInt32(kEventHotKeyPressed)
        )

        let handlerProc: EventHandlerProcPtr = { _, _, _ -> OSStatus in
            GlobalHotkeyManager.storedAction?()
            return noErr
        }

        InstallEventHandler(
            GetApplicationEventTarget(),
            handlerProc,
            1,
            &eventType,
            nil,
            &eventHandler
        )

        // Register the hotkey
        var hotKeyID = EventHotKeyID(signature: OSType(0x53594E42), id: 1) // 'SYNB'
        RegisterEventHotKey(
            keyCode,
            modifiers,
            hotKeyID,
            GetApplicationEventTarget(),
            0,
            &hotKeyRef
        )
    }

    /// Unregister the current hotkey. Called automatically on re-registration and deinit.
    func unregister() {
        if let ref = hotKeyRef {
            UnregisterEventHotKey(ref)
            hotKeyRef = nil
        }
        if let handler = eventHandler {
            RemoveEventHandler(handler)
            eventHandler = nil
        }
    }
}
#endif
```

COMPILER NOTE: If Swift refuses the closure as `EventHandlerProcPtr` (the @convention(c) bridging issue), use this alternative pattern with a top-level @convention(c) function defined outside the class:

```swift
// Top-level (outside the class) @convention(c) bridge:
private let hotkeyEventBridge: EventHandlerProcPtr = { _, _, _ -> OSStatus in
    GlobalHotkeyManager.storedAction?()
    return noErr
}
```

Then pass `hotkeyEventBridge` to InstallEventHandler. Use whichever form compiles — the behavior is identical.

**Part A — Create HotkeySettingsView.swift:**

The Settings pane shown when user presses Command+comma. It displays the current hotkey and provides a text field that captures the next key press to record a new shortcut.

```swift
//
//  HotkeySettingsView.swift
//  SyncBuds
//

#if os(macOS)
import SwiftUI
import Carbon

struct HotkeySettingsView: View {
    @AppStorage("hotkeyKeyCode") private var hotkeyKeyCode: Int = 49      // Space
    @AppStorage("hotkeyModifiers") private var hotkeyModifiers: Int = 768  // ⌘⇧

    @State private var isRecording: Bool = false

    var body: some View {
        Form {
            Section("Global Keyboard Shortcut") {
                HStack {
                    Text("Switch Headphone")
                    Spacer()
                    Button(isRecording ? "Press any key..." : shortcutDisplayString) {
                        isRecording = true
                    }
                    .buttonStyle(.bordered)
                    .onKeyPress { press in
                        guard isRecording else { return .ignored }
                        // Record the key press
                        // Note: onKeyPress gives us the SwiftUI KeyEquivalent
                        // We store the raw modifiers from NSApp.currentEvent
                        if let event = NSApp.currentEvent {
                            hotkeyKeyCode = Int(event.keyCode)
                            hotkeyModifiers = Int(event.modifierFlags.carbonFlags)
                        }
                        isRecording = false
                        return .handled
                    }
                }
                Text("Click the button above, then press your desired shortcut.")
                    .font(.caption)
                    .foregroundStyle(.secondary)
                Text("Default: ⌘⇧Space")
                    .font(.caption)
                    .foregroundStyle(.secondary)
                Button("Reset to Default") {
                    hotkeyKeyCode = 49   // Space
                    hotkeyModifiers = 768 // ⌘⇧
                }
            }
        }
        .formStyle(.grouped)
        .frame(minWidth: 380, minHeight: 160)
        .padding()
    }

    private var shortcutDisplayString: String {
        var parts: [String] = []
        let mods = UInt32(hotkeyModifiers)
        if mods & UInt32(cmdKey) != 0 { parts.append("⌘") }
        if mods & UInt32(shiftKey) != 0 { parts.append("⇧") }
        if mods & UInt32(optionKey) != 0 { parts.append("⌥") }
        if mods & UInt32(controlKey) != 0 { parts.append("⌃") }
        // Append key name for common keys
        switch hotkeyKeyCode {
        case 49: parts.append("Space")
        case 36: parts.append("Return")
        case 53: parts.append("Escape")
        default: parts.append("Key(\(hotkeyKeyCode))")
        }
        return parts.joined()
    }
}

// Extension to convert NSEvent modifier flags to Carbon modifier flags
private extension NSEvent.ModifierFlags {
    var carbonFlags: UInt32 {
        var flags: UInt32 = 0
        if contains(.command) { flags |= UInt32(cmdKey) }
        if contains(.shift) { flags |= UInt32(shiftKey) }
        if contains(.option) { flags |= UInt32(optionKey) }
        if contains(.control) { flags |= UInt32(controlKey) }
        return flags
    }
}
#endif
```

NOTE ON onKeyPress: If `onKeyPress` is not available in the macOS 26 SDK for this pattern, use `NSViewRepresentable` wrapping an NSTextField that overrides `keyDown(with:)` to capture the key code. The behavior is identical — it captures the next key press while isRecording = true. Use whichever compiles cleanly.
